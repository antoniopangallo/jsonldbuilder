# valid() → {boolean}

It's called within a node context it validates the node against a schema.
Since:
  • 1.0.0

View Source valid.js, line 43

true if node is valid, false otherwise
boolean
Example
import { create } from "jsonldbuilder"
const schemaValidator = {
  "@type": {
      opts: { alias: "type" },
      required: true,
      defaultValue: "Restaurant",
      type: "attribute"
  },
  "@context": {
      opts: { alias: "context" },
      required: true,
      defaultValue: "http://schema.org",
      type: "attribute"
  },
  "address": {
      required: true,
      type: "element",
      objectOf: {
         city: {
            required: true,
            type: "attribute"
         }
      }
   }
}
const schema = create(schemaValidator).type().context();
const address = schema.address().city("Milan");
const isAddressValid = address.valid();
const isSchemaValid = schema.valid();
